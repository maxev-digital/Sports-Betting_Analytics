### Step 3: WebSocket Service

#### File: `src/services/WebSocketService.ts`

```typescript
import {EventEmitter} from 'events';
import {BettingOpportunity} from '../types';

class WebSocketService extends EventEmitter {
  private ws: WebSocket | null = null;
  private reconnectInterval: number = 5000;
  private reconnectAttempts: number = 0;
  private maxReconnectAttempts: number = 10;
  private url: string;

  constructor(url: string) {
    super();
    this.url = url;
  }

  connect() {
    try {
      this.ws = new WebSocket(this.url);

      this.ws.onopen = () => {
        console.log('WebSocket connected');
        this.reconnectAttempts = 0;
        this.emit('connected');
      };

      this.ws.onmessage = (event) => {
        try {
          const message = JSON.parse(event.data);
          this.handleMessage(message);
        } catch (error) {
          console.error('Error parsing message:', error);
        }
      };

      this.ws.onerror = (error) => {
        console.error('WebSocket error:', error);
        this.emit('error', error);
      };

      this.ws.onclose = () => {
        console.log('WebSocket disconnected');
        this.emit('disconnected');
        this.attemptReconnect();
      };
    } catch (error) {
      console.error('Error connecting to WebSocket:', error);
      this.attemptReconnect();
    }
  }

  private handleMessage(message: any) {
    switch (message.type) {
      case 'opportunities':
        this.emit('opportunities', message.data);
        break;
      case 'new_opportunities':
        this.emit('new_opportunities', message.opportunities);
        break;
      case 'opportunity_expired':
        this.emit('opportunity_expired', message.opportunity_id);
        break;
      default:
        console.log('Unknown message type:', message.type);
    }
  }

  private attemptReconnect() {
    if (this.reconnectAttempts < this.maxReconnectAttempts) {
      this.reconnectAttempts++;
      console.log(`Reconnect attempt ${this.reconnectAttempts}/${this.maxReconnectAttempts}`);
      
      setTimeout(() => {
        this.connect();
      }, this.reconnectInterval);
    } else {
      console.error('Max reconnect attempts reached');
      this.emit('max_reconnect_attempts');
    }
  }

  send(message: any) {
    if (this.ws && this.ws.readyState === WebSocket.OPEN) {
      this.ws.send(JSON.stringify(message));
    } else {
      console.error('WebSocket is not connected');
    }
  }

  disconnect() {
    if (this.ws) {
      this.ws.close();
      this.ws = null;
    }
  }

  getOpportunities() {
    this.send({type: 'get_opportunities'});
  }
}

export default new WebSocketService('wss://api.yourdomain.com/ws/opportunities');
```

  constructor() {
    // Configure push notifications
    PushNotification.configure({
      onRegister: (token) => {
        console.log('FCM Token:', token);
        // Send token to backend
      },

      onNotification: (notification) => {
        console.log('Notification received:', notification);
        
        if (notification.userInteraction) {
          // User tapped notification
          this.handleNotificationTap(notification);
        }
      },

      permissions: {
        alert: true,
        badge: true,
        sound: true,
      },

      popInitialNotification: true,
      requestPermissions: true,
    });

    // Create notification channels (Android)
    if (Platform.OS === 'android') {
      PushNotification.createChannel(
        {
          channelId: 'opportunities',
          channelName: 'Betting Opportunities',
          channelDescription: 'Notifications for new betting opportunities',
          playSound: true,
          soundName: 'default',
          importance: 4,
          vibrate: true,
        },
        (created) => console.log(`Channel created: ${created}`)
      );
    }
  }

  sendOpportunityNotification(opportunity: BettingOpportunity) {
    const title = `${opportunity.opportunity_type.toUpperCase()} - ${opportunity.priority}`;
    const message = `${opportunity.away_team} @ ${opportunity.home_team}\nROI: ${opportunity.roi_percentage.toFixed(1)}% • ${opportunity.time_to_place}s`;

      channelId: 'opportunities',
      title: title,
      message: message,
      playSound: true,
      soundName: 'default',
      importance: 'high',
      priority: 'high',

      vibration: 300,

        opportunity_id: opportunity.opportunity_id,

    });

    const {opportunity_id} = notification.data;

### Step 5: Deep Link Service

#### File: `src/services/DeepLinkService.ts`

```typescript
import {Linking, Platform} from 'react-native';
import {BetLeg} from '../types';

class DeepLinkService {
  private sportsbookSchemes = {
    draftkings: {
      ios: 'draftkings://sportsbook',
      android: 'draftkings://sportsbook',
    },
    fanduel: {
      ios: 'fanduel://sportsbook',
      android: 'fanduel://sportsbook',
    },
    betmgm: {
      ios: 'betmgm://sports',
      android: 'betmgm://sports',
    },
    caesars: {
      ios: 'caesars://sportsbook',
      android: 'caesars://sportsbook',
    },
  };

  async openSportsbook(leg: BetLeg): Promise<boolean> {
    const scheme = this.sportsbookSchemes[leg.sportsbook as keyof typeof this.sportsbookSchemes];
    
    if (!scheme) {
      console.error(`Unknown sportsbook: ${leg.sportsbook}`);
      return false;
    }

    const deepLink = Platform.OS === 'ios' ? scheme.ios : scheme.android;
    
    try {
      // Check if app is installed
      const canOpen = await Linking.canOpenURL(deepLink);
      
      if (canOpen) {
        // Open sportsbook app
        await Linking.openURL(deepLink);
        return true;
      } else {
        // App not installed, open web version
        await Linking.openURL(leg.game_url);
        return true;
      }
    } catch (error) {
      console.error('Error opening deep link:', error);
      
      // Fallback to web
      try {
        await Linking.openURL(leg.game_url);
        return true;
      } catch (fallbackError) {
        console.error('Error opening web URL:', fallbackError);
        return false;
      }
    }
  }

  async placeBetSequence(legs: BetLeg[]): Promise<void> {
    for (const leg of legs) {
      await this.openSportsbook(leg);
      
      // Wait 8 seconds between opening apps
      await new Promise(resolve => setTimeout(resolve, 8000));
    }
  }

  async canOpenSportsbook(sportsbook: string): Promise<boolean> {
    const scheme = this.sportsbookSchemes[sportsbook as keyof typeof this.sportsbookSchemes];
    
    if (!scheme) {
      return false;
    }

    const deepLink = Platform.OS === 'ios' ? scheme.ios : scheme.android;
    
    try {
      return await Linking.canOpenURL(deepLink);
    } catch {
      return false;
    }
  }
}

export default new DeepLinkService();
```

  const [opportunities, setOpportunities] = useState<BettingOpportunity[]>([]);
  const [stats, setStats] = useState<UserStats>({
    total_opportunities: 0,
    opportunities_today: 0,
    total_profit: 0,
    roi_percentage: 0,
    win_rate: 0,
  });
  const [connected, setConnected] = useState(false);
  const [refreshing, setRefreshing] = useState(false);

  useEffect(() => {
    // Connect to WebSocket
    WebSocketService.connect();

    // Listen for connection events
    WebSocketService.on('connected', () => {
      setConnected(true);
      WebSocketService.getOpportunities();
    });

    WebSocketService.on('disconnected', () => {
      setConnected(false);

    WebSocketService.on('opportunities', (data: BettingOpportunity[]) => {
      setOpportunities(data);

          NotificationService.sendOpportunityNotification(opp);
        }
      });

    return () => {
      WebSocketService.removeAllListeners();
    };
  }, []);

  const onRefresh = () => {
    setRefreshing(true);
    WebSocketService.getOpportunities();
    setTimeout(() => setRefreshing(false), 2000);
