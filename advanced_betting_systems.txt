                    game_date,
                    opening_total,

                'days': days,
                'games_analyzed': len(clv_results),
                'avg_clv': round(avg_clv, 2),
                'closing_line_win_rate': round(win_rate * 100, 1)
            }

```python
"""Collection of advanced betting strategies"""

from typing import Dict, List, Optional
from dataclasses import dataclass
from enum import Enum

class StrategyType(str, Enum):
    PACE_BASED = "pace_based"
    MOMENTUM = "momentum"
    REGRESSION = "regression"
    H2H_TRENDS = "h2h_trends"
    LINE_MOVEMENT = "line_movement"
    SITUATIONAL = "situational"
    REFEREE = "referee"
    FATIGUE = "fatigue"
    GARBAGE_TIME = "garbage_time"
    LINEUP_IMPACT = "lineup_impact"
    PLAYER_PROPS = "player_props"

@dataclass
class StrategySignal:
    """Output from a betting strategy"""
    strategy_type: StrategyType
    recommendation: str  # "OVER", "UNDER", "HOME", "AWAY", etc.
    confidence: float  # 0.0 to 1.0
    edge: float  # Projected edge in points/units
    reasoning: str
    supporting_data: Dict
    weight: float = 1.0  # Weight in ensemble model

class StrategyEngine:
    """Base class for all betting strategies"""
    
    def __init__(self, stats_db):
        self.stats_db = stats_db
        self.strategy_type = None
    
    async def analyze(
        self,
        game,
        odds: List,
        context
    ) -> Optional[StrategySignal]:
        """Analyze game and return signal if opportunity exists"""
        raise NotImplementedError

class PaceBasedStrategy(StrategyEngine):
    """Pace-based total projections"""
    
    strategy_type = StrategyType.PACE_BASED
    
    async def analyze(self, game, odds, context):
        # Get team pace stats
        home_stats = await self.stats_db.get_team_stats(
            game.home_team.name, game.sport
        )
        away_stats = await self.stats_db.get_team_stats(
            game.away_team.name, game.sport
        )
        
        # Calculate expected pace for matchup
        expected_pace = (home_stats.pace + away_stats.pace) / 2
        
        # Adjust for injuries, rest, etc.
        if context.home_injury_impact_score > 7:
            expected_pace *= 0.95  # Injuries slow pace
        if context.away_injury_impact_score > 7:
            expected_pace *= 0.95
        
        # Project total
        home_pts_per_100 = home_stats.offensive_rating
        away_pts_per_100 = away_stats.offensive_rating
        
        projected_total = (
            (home_pts_per_100 + away_pts_per_100) * expected_pace / 100
        )
        
        # Compare to market
        market_total = odds[0].total_line if odds else 220.0
        edge = abs(projected_total - market_total)
        
        if edge >= 3.0:  # 3+ point edge
            return StrategySignal(
                strategy_type=self.strategy_type,
                recommendation="OVER" if projected_total > market_total else "UNDER",
                confidence=min(edge / 10, 1.0),
                edge=edge,
                reasoning=f"Pace-based model projects {projected_total:.1f}, market at {market_total}",
                supporting_data={
                    'expected_pace': expected_pace,
                    'home_off_rating': home_pts_per_100,
                    'away_off_rating': away_pts_per_100,
                    'projected_total': projected_total,
                    'market_total': market_total
                },
                weight=1.0
            )
        
        return None

class RegressionStrategy(StrategyEngine):
    """Regression to the mean strategy"""
    
    strategy_type = StrategyType.REGRESSION
    
    async def analyze(self, game, odds, context):
        home_stats = await self.stats_db.get_team_stats(
            game.home_team.name, game.sport
        )
        away_stats = await self.stats_db.get_team_stats(
            game.away_team.name, game.sport
        )
        
        # Check if teams are running hot/cold
        home_last5_vs_season = home_stats.last_5_ppg - home_stats.points_per_game
        away_last5_vs_season = away_stats.last_5_ppg - away_stats.points_per_game
        
        # If both teams hot (scoring 5+ more than average), expect regression
        if home_last5_vs_season > 5 and away_last5_vs_season > 5:
            market_total = odds[0].total_line if odds else 220.0
            expected_regression = 6.0  # Expect 6 point drop
            
            return StrategySignal(
                strategy_type=self.strategy_type,
                recommendation="UNDER",
                confidence=0.65,
                edge=expected_regression,
                reasoning=f"Both teams hot (Home +{home_last5_vs_season:.1f}, Away +{away_last5_vs_season:.1f}), expect regression",
                supporting_data={
                    'home_deviation': home_last5_vs_season,
                    'away_deviation': away_last5_vs_season,
                    'expected_regression': expected_regression
                },
                weight=0.8
            )
        
        # If both teams cold, expect bounce-back
        if home_last5_vs_season < -5 and away_last5_vs_season < -5:
            return StrategySignal(
                strategy_type=self.strategy_type,
                recommendation="OVER",
                confidence=0.60,
                edge=5.0,
                reasoning=f"Both teams cold, expect regression to mean",
                supporting_data={
                    'home_deviation': home_last5_vs_season,
                    'away_deviation': away_last5_vs_season
                },
                weight=0.7
            )
        
        return None

class FatigueStrategy(StrategyEngine):
    """Back-to-back and rest-based strategy"""
    
    strategy_type = StrategyType.FATIGUE
    
    async def analyze(self, game, odds, context):
        # Back-to-back games tend to go UNDER
        if context.home_back_to_back or context.away_back_to_back:
            fatigue_factor = 0
            reasoning_parts = []
            
            if context.home_back_to_back:
                fatigue_factor += 2.5
                reasoning_parts.append(f"{game.home_team.name} on back-to-back")
            
            if context.away_back_to_back:
                fatigue_factor += 2.5
                reasoning_parts.append(f"{game.away_team.name} on back-to-back")
            
            # Add travel fatigue
            if context.away_travel_distance > 1500:  # Cross-country
                fatigue_factor += 1.5
                reasoning_parts.append(f"Away team traveled {context.away_travel_distance:.0f} miles")
            
            if fatigue_factor >= 3.0:
                return StrategySignal(
                    strategy_type=self.strategy_type,
                    recommendation="UNDER",
                    confidence=0.70,
                    edge=fatigue_factor,
                    reasoning=" + ".join(reasoning_parts),
                    supporting_data={
                        'home_rest': context.home_rest_days,
                        'away_rest': context.away_rest_days,
                        'home_b2b': context.home_back_to_back,
                        'away_b2b': context.away_back_to_back,
                        'travel_distance': context.away_travel_distance
                    },
                    weight=0.9
                )
        
        # Rest advantage (3+ days vs 0-1 days)
        rest_diff = abs(context.home_rest_days - context.away_rest_days)
        if rest_diff >= 2:
            rested_team = "home" if context.home_rest_days > context.away_rest_days else "away"
            
            return StrategySignal(
                strategy_type=self.strategy_type,
                recommendation=f"{rested_team.upper()}_SPREAD",
                confidence=0.65,
                edge=rest_diff * 1.5,
                reasoning=f"{rested_team.title()} team has {rest_diff} day rest advantage",
                supporting_data={
                    'rest_differential': rest_diff,
                    'advantage_team': rested_team
                },
                weight=0.8
            )
        
        return None

class RefereeStrategy(StrategyEngine):
    """Referee tendency-based strategy"""
    
    strategy_type = StrategyType.REFEREE
    
    async def analyze(self, game, odds, context):
        if not context.referee_name:
            return None
        
        ref_tendency = context.referee_over_under_tendency
        if ref_tendency is None:
            return None
        
        # If referee has strong over/under tendency (>55% or <45%)
        if ref_tendency > 55:
            return StrategySignal(
                strategy_type=self.strategy_type,
                recommendation="OVER",
                confidence=0.60,
                edge=2.0,
                reasoning=f"Ref {context.referee_name} games go OVER {ref_tendency}% of time",
                supporting_data={
                    'referee': context.referee_name,
                    'over_percentage': ref_tendency,
                    'avg_fouls': context.referee_avg_fouls_called
                },
                weight=0.6
            )
        
        elif ref_tendency < 45:
            return StrategySignal(
                strategy_type=self.strategy_type,
                recommendation="UNDER",
                confidence=0.60,
                edge=2.0,
                reasoning=f"Ref {context.referee_name} games go UNDER {100-ref_tendency}% of time",
                supporting_data={
                    'referee': context.referee_name,
                    'over_percentage': ref_tendency,
                    'avg_fouls': context.referee_avg_fouls_called
                },
                weight=0.6
            )
        
        return None

class GarbageTimeStrategy(StrategyEngine):
    """Detect and adjust for garbage time scenarios"""
    
    strategy_type = StrategyType.GARBAGE_TIME
    
    async def analyze(self, game, odds, context):
        # Only applies to live games
        if game.status != "live":
            return None
        
        # Check if game is a blowout
        if not (game.home_team.score and game.away_team.score):
            return None
        
        score_diff = abs(game.home_team.score - game.away_team.score)
        
        # NBA: 15+ point lead in Q4 = garbage time likely
        if game.sport == "basketball_nba":
            if game.period >= 4 and score_diff >= 15:
                return StrategySignal(
                    strategy_type=self.strategy_type,
                    recommendation="OVER",
                    confidence=0.75,
                    edge=3.0,
                    reasoning=f"Garbage time expected ({score_diff} point lead in Q4)",
                    supporting_data={
                        'score_differential': score_diff,
                        'period': game.period,
                        'time_remaining': game.time_remaining
                    },
                    weight=1.0
                )
        
        # NFL: 3+ TD lead in 4th quarter
        elif game.sport == "americanfootball_nfl":
            if game.period >= 4 and score_diff >= 21:
                return StrategySignal(
                    strategy_type=self.strategy_type,
                    recommendation="OVER",
                    confidence=0.70,
                    edge=4.0,
                    reasoning=f"Prevent defense + hurry-up offense likely",
                    supporting_data={
                        'score_differential': score_diff,
                        'period': game.period
                    },
                    weight=0.9
                )
        
        return None

class MomentumStrategy(StrategyEngine):
    """Recent scoring momentum strategy"""
    
    strategy_type = StrategyType.MOMENTUM
    
    async def analyze(self, game, odds, context):
        # Only for live games
        if game.status != "live":
            return None
        
        home_stats = await self.stats_db.get_team_stats(
            game.home_team.name, game.sport
        )
        away_stats = await self.stats_db.get_team_stats(
            game.away_team.name, game.sport
        )
        
        # Calculate current pace vs season average
        if game.period >= 2:  # Need at least some game data
            current_total = game.home_team.score + game.away_team.score
            expected_total = home_stats.avg_total
            
            # If scoring significantly faster than average
            pace_factor = current_total / expected_total
            
            if pace_factor > 1.15:  # 15% faster pace
                return StrategySignal(
                    strategy_type=self.strategy_type,
                    recommendation="OVER",
                    confidence=0.70,
                    edge=3.5,
                    reasoning=f"Current pace {pace_factor:.1%} of expected, momentum suggests OVER",
                    supporting_data={
                        'current_total': current_total,
                        'expected_total': expected_total,
                        'pace_factor': pace_factor
                    },
                    weight=0.85
                )
            
            elif pace_factor < 0.85:  # 15% slower pace
                return StrategySignal(
                    strategy_type=self.strategy_type,
                    recommendation="UNDER",
                    confidence=0.70,
                    edge=3.5,
                    reasoning=f"Current pace {pace_factor:.1%} of expected, momentum suggests UNDER",
                    supporting_data={
                        'current_total': current_total,
                        'expected_total': expected_total,
                        'pace_factor': pace_factor
                    },
                    weight=0.85
                )
        
        return None

"""Combine multiple strategies into unified predictions"""

from typing import List, Dict
import numpy as np

class EnsembleModel:
    """Combine multiple betting strategies with weighted voting"""
    
    def __init__(self, stats_db):
        self.stats_db = stats_db
        
        # Initialize all strategies
        self.strategies = [
            PaceBasedStrategy(stats_db),
            RegressionStrategy(stats_db),
            FatigueStrategy(stats_db),
            RefereeStrategy(stats_db),
            GarbageTimeStrategy(stats_db),
            MomentumStrategy(stats_db),
        ]
    
    async def analyze_game(
        self,
        game,
        odds: List,
        context
    ) -> Dict:
        """Run all strategies and combine results"""
        
        signals = []
        
        # Run each strategy
        for strategy in self.strategies:
            try:
                signal = await strategy.analyze(game, odds, context)
                if signal:
                    signals.append(signal)
            except Exception as e:
                print(f"Strategy {strategy.strategy_type} failed: {e}")
                continue
        
        if not signals:
            return {
                'recommendation': None, 
                'confidence': 0, 
                'edge': 0,
                'expected_value': 0,
                'is_positive_ev': False
            }
        
        # Combine signals using weighted voting
        combined = self._combine_signals(signals, odds)
        
        return combined
    
    def _combine_signals(
        self, 
        signals: List,
        odds: List
    ) -> Dict:
        """Combine multiple signals into unified recommendation"""
        
        # Group signals by recommendation type
        over_signals = [s for s in signals if s.recommendation == "OVER"]
        under_signals = [s for s in signals if s.recommendation == "UNDER"]
        
        # Calculate weighted scores
        over_score = sum(s.confidence * s.weight for s in over_signals)
        under_score = sum(s.confidence * s.weight for s in under_signals)
        
        # Determine recommendation
        if over_score > under_score and over_score >= 1.0:
            recommendation = "OVER"
            confidence = min(over_score / (over_score + under_score), 1.0)
            supporting_strategies = [s.strategy_type.value for s in over_signals]
            avg_edge = np.mean([s.edge for s in over_signals])
        
        elif under_score > over_score and under_score >= 1.0:
            recommendation = "UNDER"
            confidence = min(under_score / (over_score + under_score), 1.0)
            supporting_strategies = [s.strategy_type.value for s in under_signals]
            avg_edge = np.mean([s.edge for s in under_signals])
        
        else:
            return {
                'recommendation': None, 
                'confidence': 0, 
                'edge': 0,
                'expected_value': 0,
                'is_positive_ev': False
            }
        
        # Calculate expected value (EV)
        market_total = odds[0].total_line if odds else 220.0
        market_odds = odds[0].over_price if recommendation == "OVER" else odds[0].under_price
        
        # Convert American odds to implied probability
        if market_odds < 0:
            implied_prob = abs(market_odds) / (abs(market_odds) + 100)
        else:
            implied_prob = 100 / (market_odds + 100)
        
        # Our true probability (confidence adjusted)
        true_prob = confidence
        
        # Calculate EV
        ev = (true_prob * 100) - (implied_prob * 100)
        
        return {
            'recommendation': recommendation,
            'confidence': round(confidence * 100, 1),  # Convert to percentage
            'edge': round(avg_edge, 1),
            'expected_value': round(ev, 2),
            'strategies_agreeing': len(supporting_strategies),
            'strategy_breakdown': supporting_strategies,
            'signals': [
                {
                    'type': s.strategy_type.value,
                    'rec': s.recommendation,
                    'confidence': round(s.confidence * 100, 1),
                    'reasoning': s.reasoning
                }
                for s in signals
            ],
            'market_total': market_total,
            'market_odds': market_odds,
            'implied_probability': round(implied_prob * 100, 1),
            'true_probability': round(true_prob * 100, 1),
            'is_positive_ev': ev > 0
        }
```

## 💰 Expected Value Calculator

```python
"""Expected Value calculations for betting opportunities"""

from typing import Dict, Optional

class EVCalculator:
    """Calculate expected value for betting opportunities"""
    
    @staticmethod
    def american_to_decimal(odds: int) -> float:
        """Convert American odds to decimal"""
        if odds > 0:
            return (odds / 100) + 1
        else:
            return (100 / abs(odds)) + 1
    
    @staticmethod
    def american_to_implied_prob(odds: int) -> float:
        """Convert American odds to implied probability"""
        if odds < 0:
            return abs(odds) / (abs(odds) + 100)
        else:
            return 100 / (odds + 100)
    
    @staticmethod
    def calculate_ev(
        true_probability: float,
        market_odds: int,
        bet_amount: float = 100
    ) -> Dict:
        """
        Calculate expected value of a bet
        
        Args:
            true_probability: Our model's probability (0.0 to 1.0)
            market_odds: American odds (-110, +150, etc.)
            bet_amount: Size of bet in dollars
        
        Returns:
            Dict with EV calculations
        """
        # Convert odds to implied probability
        implied_prob = EVCalculator.american_to_implied_prob(market_odds)
        
        # Calculate payout
        decimal_odds = EVCalculator.american_to_decimal(market_odds)
        potential_profit = bet_amount * (decimal_odds - 1)
        
        # Calculate EV
        win_amount = potential_profit
        lose_amount = bet_amount
        
        ev_dollars = (true_probability * win_amount) - ((1 - true_probability) * lose_amount)
        ev_percentage = (ev_dollars / bet_amount) * 100
        
        # Edge
        edge = true_probability - implied_prob
        edge_percentage = edge * 100
        
        return {
            'expected_value_dollars': round(ev_dollars, 2),
            'expected_value_percentage': round(ev_percentage, 2),
            'edge': round(edge, 4),
            'edge_percentage': round(edge_percentage, 2),
            'true_probability': round(true_probability * 100, 1),
            'implied_probability': round(implied_prob * 100, 1),
            'is_positive_ev': ev_dollars > 0,
            'potential_profit': round(potential_profit, 2),
            'risk_amount': bet_amount
        }
    
    @staticmethod
    def kelly_criterion(
        edge: float,
        odds: int,
        bankroll: float,
        kelly_fraction: float = 0.25
    ) -> Dict:
        """
        Calculate optimal bet size using Kelly Criterion
        
        Args:
            edge: Our edge (true_prob - implied_prob)
            odds: American odds
            bankroll: Total bankroll
            kelly_fraction: Fraction of Kelly to use (0.25 = quarter Kelly)
        
        Returns:
            Dict with bet sizing recommendations
        """
        # Convert to decimal odds
        decimal_odds = EVCalculator.american_to_decimal(odds)
        
        # Kelly formula: (edge * decimal_odds) / (decimal_odds - 1)
        if edge <= 0:
            return {
                'kelly_bet_size': 0,
                'recommendation': 'NO BET',
                'reason': 'No edge detected'
            }
        
        full_kelly = (edge * decimal_odds) / (decimal_odds - 1)
        
        # Apply fraction
        fractional_kelly = full_kelly * kelly_fraction
        
        # Calculate bet size
        bet_size = bankroll * fractional_kelly
        
        # Cap at reasonable maximums
        max_bet = bankroll * 0.05  # Never bet more than 5% of bankroll
        bet_size = min(bet_size, max_bet)
        
        return {
            'full_kelly_percentage': round(full_kelly * 100, 2),
            'fractional_kelly_percentage': round(fractional_kelly * 100, 2),
            'recommended_bet_size': round(bet_size, 2),
            'recommended_units': round(bet_size / 100, 2),  # Assuming $100 = 1 unit
            'kelly_fraction_used': kelly_fraction,
            'bankroll': bankroll,
            'max_bet_allowed': max_bet
        }
```
